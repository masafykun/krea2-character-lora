#!/bin/bash
set -e
cd /mnt/data/krea2-lora/musubi-tuner
source .venv/bin/activate
export PYTHONUNBUFFERED=1
M=/mnt/data/krea2-lora/models
DS=/mnt/data/krea2-lora/dataset.toml
OUT=/mnt/data/krea2-lora/output

# 1) latent事前キャッシュ（VAE使用）
python src/musubi_tuner/krea2_cache_latents.py \
  --dataset_config $DS --vae $M/qwen_image_vae.safetensors

# 2) テキストエンコーダ出力 事前キャッシュ（Qwen3-VL）
python src/musubi_tuner/krea2_cache_text_encoder_outputs.py \
  --dataset_config $DS --text_encoder $M/qwen3vl_4b_bf16.safetensors --batch_size 1

# 3) 学習（RAWで学習、fp8_scaled+block_swap26で12GBに収める）
accelerate launch --num_cpu_threads_per_process 1 --mixed_precision bf16 \
  src/musubi_tuner/krea2_train_network.py \
  --dit $M/krea2_raw.safetensors \
  --vae $M/qwen_image_vae.safetensors \
  --dataset_config $DS \
  --sdpa --mixed_precision bf16 \
  --fp8_base --fp8_scaled --blocks_to_swap 26 \
  --timestep_sampling shift --weighting_scheme none --discrete_flow_shift 2.5 \
  --network_module networks.lora_krea2 --network_dim 32 --network_alpha 32 \
  --optimizer_type adamw --learning_rate 1e-4 \
  --max_train_epochs 10 --save_every_n_epochs 2 \
  --gradient_checkpointing --seed 42 \
  --output_dir $OUT --output_name masafee_krea2 \
  --logging_dir /mnt/data/krea2-lora/logs
echo "=== TRAIN DONE exit=$? ==="
